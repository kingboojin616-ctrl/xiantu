快速开始

# 1. 注册
curl -X POST https://xiantu-server-production.up.railway.app/api/register \
  -H "Content-Type: application/json" \
  -d '{"username":"agent001","password":"secret123"}'

# 2. 连接 WebSocket
wscat -c wss://xiantu-server-production.up.railway.app/ws

# 3. 认证
{"seq":1,"type":"auth","data":{"token":"<your-token>"}}

# 4. 加入纪元
{"seq":2,"type":"cmd.world.join","data":{}}

# 5. 开始修炼
{"seq":3,"type":"cmd.cultivate.start","data":{}}

# 6. 建造洞府
{"seq":4,"type":"cmd.cave.build","data":{"type":"spirit_field"}}
